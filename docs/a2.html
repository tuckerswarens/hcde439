Here is my code snippet for the blinking lED's: 


    //define the button to pin 2
const int buttonPin = 2;
//constant ints don't change. I'm adding in which led corresponds to each pin
//fadeing led goes with pin 9
const int LEDFade = 9;
//led 2 goes with pin 10
const int LED2 = 10;
//led 3 goes with pin 11
const int LED3 = 11;

//stores the value of the state of the button either on or off
int buttonState = 0;

// the setup function runs once when you press reset or power the board
void setup() {
  // initialize all the pins with leds as outputs
  //pin 9 is an output 
  pinMode(LEDFade, OUTPUT);
  //pin 10 is an output 
  pinMode(LED2, OUTPUT);
  //pin 11 is and output 
  pinMode(LED3, OUTPUT),

//initialize the button pin as an input
  pinMode(buttonPin, INPUT);
}

// the loop function runs over and over again forever
void loop() {
  //reads the current state of the button
  buttonState = digitalRead(buttonPin);
//checks if button is pressed and if it is runs the code inside the if statment
  if (buttonState == HIGH) {
    //turn on led 2 which is red
    digitalWrite(LED2, HIGH);
    
    //fade led 1 up using a for loop increasing brightness from 0
    for (int brightness = 0; brightness <= 255; brightness +=5) {
      //turn led on pin 9 to the brightness value from the for loop
      analogWrite(LEDFade, brightness);
      //delay 0.02 seconds
      delay(20);
    }
    
    // fade led 1 down using a for loop deacreasing the brightness from 255 in incriments of 5
    for (int brightness = 255; brightness >= 0; brightness -= 5) {
      //turn led on pin 9 to the brightness value from the for loop
      analogWrite(LEDFade, brightness);
      //delay 0.02 seconds
      delay(20);
    }
    
    //turn led 3 on then off then on then off again with 0.1 second delays
    //turn led 3 on
    digitalWrite(LED3, HIGH);
    //delay 0.1 seconds
    delay(100);
    //turn led 3 off
    digitalWrite(LED3, LOW);
    //delay 0.1 seconds
    delay(100);
    //turn led 3 on
    digitalWrite(LED3, HIGH);
    //delay 0.1 seconds
    delay(100);
    //turn led 3 off
    digitalWrite(LED3, LOW);

    //turn all the leds off if the button is not pressed 
  } else {
    //turn led 2 off
    digitalWrite(LED2, LOW);
    //turn led 3 off
    digitalWrite(LED3, LOW);
    //turn ledfade off
    analogWrite(LEDFade, 0);
  }

}
