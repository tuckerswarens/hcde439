Here is my code snippet for the arduino: 


    // Define joystick X-axis pin
int joyX = A0;

// Define joystick Y-axis pin
int joyY = A1;

// Define joystick button pin
int joyButton = 2;

// Define LED output pin (PWM capable)
int ledPin = 9;

// Variable to store incoming LED brightness from webpage
int ledBrightness = 0;

void setup() {

  // Start serial communication at 9600 baud
  Serial.begin(9600);

  // Set joystick button as input
  pinMode(joyButton, INPUT_PULLUP);

  // Set LED pin as output
  pinMode(ledPin, OUTPUT);
}

void loop() {

  // Read joystick X value (0-1023)
  int xValue = analogRead(joyX);

  // Read joystick Y value (0-1023)
  int yValue = analogRead(joyY);

  // Read joystick button state (0 or 1)
  int buttonState = digitalRead(joyButton);

  // Send X value over serial
  Serial.print(xValue);

  // Send comma separator
  Serial.print(",");

  Serial.print(yValue);

  // Send Y value over serial
  Serial.print(",");

  // Send button value
  Serial.println(buttonState);

  // Check if data is available from webpage
  if (Serial.available() > 0) {

    // Read brightness value sent from webpage
    ledBrightness = Serial.parseInt();

    // Write brightness to LED (PWM 0-255)
    analogWrite(ledPin, ledBrightness);
  }

  // Small delay to stabilize communication
  delay(20);
}
